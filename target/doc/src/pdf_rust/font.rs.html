1//! フォント関連: 標準 14 フォントのメトリクスと WinAnsi エンコーディング。
2//!
3//! PDF ビューアが必ず内蔵する「標準 14 フォント」（§9.6.2.2）は、
4//! フォントファイルを埋め込まずに利用できる。本モジュールはそのうち
5//! 主要書体の文字幅テーブル（AFM 由来、1000 分の 1 em 単位）を持ち、
6//! テキスト幅の計算（レイアウト補助）に使う。
7
8/// 標準 14 フォント。
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10pub enum StandardFont {
11    Helvetica,
12    HelveticaBold,
13    HelveticaOblique,
14    HelveticaBoldOblique,
15    TimesRoman,
16    TimesBold,
17    TimesItalic,
18    TimesBoldItalic,
19    Courier,
20    CourierBold,
21    CourierOblique,
22    CourierBoldOblique,
23    Symbol,
24    ZapfDingbats,
25}
26
27impl StandardFont {
28    /// PDF の `/BaseFont` 名。
29    pub fn base_font(&self) -> &'static str {
30        match self {
31            StandardFont::Helvetica => "Helvetica",
32            StandardFont::HelveticaBold => "Helvetica-Bold",
33            StandardFont::HelveticaOblique => "Helvetica-Oblique",
34            StandardFont::HelveticaBoldOblique => "Helvetica-BoldOblique",
35            StandardFont::TimesRoman => "Times-Roman",
36            StandardFont::TimesBold => "Times-Bold",
37            StandardFont::TimesItalic => "Times-Italic",
38            StandardFont::TimesBoldItalic => "Times-BoldItalic",
39            StandardFont::Courier => "Courier",
40            StandardFont::CourierBold => "Courier-Bold",
41            StandardFont::CourierOblique => "Courier-Oblique",
42            StandardFont::CourierBoldOblique => "Courier-BoldOblique",
43            StandardFont::Symbol => "Symbol",
44            StandardFont::ZapfDingbats => "ZapfDingbats",
45        }
46    }
47
48    /// ASCII 文字の幅（1000 分の 1 em 単位）。表にない文字は 500 を返す。
49    pub fn char_width(&self, c: char) -> u16 {
50        let idx = c as usize;
51        if !(0x20..=0x7E).contains(&idx) {
52            return 500;
53        }
54        let i = idx - 0x20;
55        match self {
56            StandardFont::Courier
57            | StandardFont::CourierBold
58            | StandardFont::CourierOblique
59            | StandardFont::CourierBoldOblique => 600,
60            StandardFont::Helvetica | StandardFont::HelveticaOblique => HELVETICA_WIDTHS[i],
61            StandardFont::HelveticaBold | StandardFont::HelveticaBoldOblique => {
62                HELVETICA_BOLD_WIDTHS[i]
63            }
64            StandardFont::TimesRoman => TIMES_ROMAN_WIDTHS[i],
65            // 簡易版: イタリック・ボールド系は Roman の幅で近似
66            StandardFont::TimesBold | StandardFont::TimesItalic | StandardFont::TimesBoldItalic => {
67                TIMES_ROMAN_WIDTHS[i]
68            }
69            StandardFont::Symbol | StandardFont::ZapfDingbats => 500,
70        }
71    }
72
73    /// 文字列をフォントサイズ `size` で描画したときの幅（ポイント単位）。
74    pub fn measure_text(&self, text: &str, size: f64) -> f64 {
75        let units: u64 = text.chars().map(|c| self.char_width(c) as u64).sum();
76        units as f64 * size / 1000.0
77    }
78}

119/// WinAnsiEncoding（CP1252 相当, §Annex D.2）のバイト → Unicode 変換。
120///
121/// 0x80..0x9F の領域は CP1252 固有のマッピングを持つ。
122pub fn winansi_to_char(b: u8) -> char {
123    match b {
124        0x80 => '€',
125        0x82 => '‚',
126        0x83 => 'ƒ',
127        0x84 => '„',
128        0x85 => '…',
129        0x86 => '†',
130        0x87 => '‡',
131        0x88 => 'ˆ',
132        0x89 => '‰',
133        0x8A => 'Š',
134        0x8B => '‹',
135        0x8C => 'Œ',
136        0x8E => 'Ž',
137        0x91 => '\u{2018}', // '
138        0x92 => '\u{2019}', // '
139        0x93 => '\u{201C}', // "
140        0x94 => '\u{201D}', // "
141        0x95 => '•',
142        0x96 => '–',
143        0x97 => '—',
144        0x98 => '˜',
145        0x99 => '™',
146        0x9A => 'š',
147        0x9B => '›',
148        0x9C => 'œ',
149        0x9E => 'ž',
150        0x9F => 'Ÿ',
151        // 未定義コード（0x81 など）は U+FFFD ではなく空白に近い扱い
152        0x81 | 0x8D | 0x8F | 0x90 | 0x9D => '\u{FFFD}',
153        // それ以外は Latin-1 と同一
154        b => b as char,
155    }
156}

158/// Unicode 文字 → WinAnsi バイト（書き込み用）。表せない文字は `None`。
159pub fn char_to_winansi(c: char) -> Option<u8> {
160    let cp = c as u32;
161    match c {
162        '€' => Some(0x80),
163        '‚' => Some(0x82),
164        'ƒ' => Some(0x83),
165        '„' => Some(0x84),
166        '…' => Some(0x85),
167        '†' => Some(0x86),
168        '‡' => Some(0x87),
169        'ˆ' => Some(0x88),
170        '‰' => Some(0x89),
171        'Š' => Some(0x8A),
172        '‹' => Some(0x8B),
173        'Œ' => Some(0x8C),
174        'Ž' => Some(0x8E),
175        '\u{2018}' => Some(0x91),
176        '\u{2019}' => Some(0x92),
177        '\u{201C}' => Some(0x93),
178        '\u{201D}' => Some(0x94),
179        '•' => Some(0x95),
180        '–' => Some(0x96),
181        '—' => Some(0x97),
182        '˜' => Some(0x98),
183        '™' => Some(0x99),
184        'š' => Some(0x9A),
185        '›' => Some(0x9B),
186        'œ' => Some(0x9C),
187        'ž' => Some(0x9E),
188        'Ÿ' => Some(0x9F),
189        _ if cp < 0x80 => Some(cp as u8),
190        _ if (0xA0..=0xFF).contains(&cp) => Some(cp as u8),
191        _ => None,
192    }
193}

195#[cfg(test)]
196mod tests {
197    use super::*;
198
199    #[test]
200    fn widths() {
201        assert_eq!(StandardFont::Helvetica.char_width('A'), 667);
202        assert_eq!(StandardFont::Helvetica.char_width(' '), 278);
203        assert_eq!(StandardFont::Courier.char_width('W'), 600);
204        let w = StandardFont::Helvetica.measure_text("AA", 10.0);
205        assert!((w - 13.34).abs() < 1e-9);
206    }
207
208    #[test]
209    fn winansi_roundtrip() {
210        for c in ['A', 'é', '€', '\u{201C}', '~'] {
211            let b = char_to_winansi(c).unwrap();
212            assert_eq!(winansi_to_char(b), c);
213        }
214        assert_eq!(char_to_winansi('あ'), None);
215    }
