1//! PDF シリアライザ。
2//!
3//! オブジェクトをファイル形式のバイト列に変換し、ドキュメント全体を
4//! 古典的 xref テーブル形式で書き出す（§7.5）。
5
6use crate::error::Result;
7use crate::object::{Dictionary, Object, StringFormat};

9/// オブジェクト 1 つをシリアライズして `out` に追記する。
10pub fn write_object(out: &mut Vec<u8>, obj: &Object) {
11    match obj {
12        Object::Null => out.extend_from_slice(b"null"),
13        Object::Boolean(true) => out.extend_from_slice(b"true"),
14        Object::Boolean(false) => out.extend_from_slice(b"false"),
15        Object::Integer(i) => out.extend_from_slice(i.to_string().as_bytes()),
16        Object::Real(v) => out.extend_from_slice(format_real(*v).as_bytes()),
17        Object::String(s, fmt) => write_string(out, s, *fmt),
18        Object::Name(n) => write_name(out, n),
19        Object::Array(items) => {
20            out.push(b'[');
21            for (i, item) in items.iter().enumerate() {
22                if i > 0 {
23                    out.push(b' ');
24                }
25                write_object(out, item);
26            }
27            out.push(b']');
28        }
29        Object::Dictionary(d) => write_dict(out, d),
30        Object::Stream(s) => {
31            write_dict(out, &s.dict);
32            out.extend_from_slice(b"\nstream\n");
33            out.extend_from_slice(&s.data);
34            out.extend_from_slice(b"\nendstream");
35        }
36        Object::Reference((n, g)) => {
37            out.extend_from_slice(format!("{n} {g} R").as_bytes());
38        }
39    }
40}

42/// 実数を PDF 表記でフォーマットする（末尾の 0 を除去）。
43pub fn format_real(v: f64) -> String {
44    if !v.is_finite() {
45        return "0".into(); // PDF に NaN/inf は書けない
46    }
47    if v == v.trunc() && v.abs() < 1e15 {
48        return format!("{}", v as i64);
49    }
50    let mut s = format!("{v:.6}");
51    while s.ends_with('0') {
52        s.pop();
53    }
54    if s.ends_with('.') {
55        s.pop();
56    }
57    s
58}

60fn write_dict(out: &mut Vec<u8>, d: &Dictionary) {
61    out.extend_from_slice(b"<<");
62    for (i, (k, v)) in d.iter().enumerate() {
63        if i > 0 {
64            out.push(b' ');
65        }
66        write_name(out, k);
67        // 名前キーと値の区切り（値が名前や区切り文字で始まる場合は省略可能だが常に空ける）
68        out.push(b' ');
69        write_object(out, v);
70    }
71    out.extend_from_slice(b">>");
72}
73
74fn write_name(out: &mut Vec<u8>, name: &str) {
75    out.push(b'/');
76    for &b in name.as_bytes() {
77        // 通常文字以外と '#'、非印字文字は #xx でエスケープ（§7.3.5）
78        if crate::lexer::is_regular(b) && b != b'#' && (0x21..=0x7E).contains(&b) {
79            out.push(b);
80        } else {
81            out.extend_from_slice(format!("#{b:02X}").as_bytes());
82        }
83    }
84}

86fn write_string(out: &mut Vec<u8>, s: &[u8], fmt: StringFormat) {
87    // 非印字バイトが多い場合は 16 進形式を選ぶ
88    let nonprintable = s.iter().filter(|&&b| !(0x20..=0x7E).contains(&b)).count();
89    let use_hex = fmt == StringFormat::Hexadecimal || (!s.is_empty() && nonprintable * 2 > s.len());
90    if use_hex {
91        out.push(b'<');
92        for &b in s {
93            out.extend_from_slice(format!("{b:02X}").as_bytes());
94        }
95        out.push(b'>');
96    } else {
97        out.push(b'(');
98        for &b in s {
99            match b {
100                b'(' | b')' | b'\\' => {
101                    out.push(b'\\');
102                    out.push(b);
103                }
104                b'\n' => out.extend_from_slice(b"\\n"),
105                b'\r' => out.extend_from_slice(b"\\r"),
106                b'\t' => out.extend_from_slice(b"\\t"),
107                0x08 => out.extend_from_slice(b"\\b"),
108                0x0C => out.extend_from_slice(b"\\f"),
109                b if b < 0x20 => out.extend_from_slice(format!("\\{b:03o}").as_bytes()),
110                b => out.push(b),
111            }
112        }
113        out.push(b')');
114    }
115}

117/// ドキュメント全体をシリアライズする。
118///
119/// すべてのオブジェクトを非圧縮の間接オブジェクトとして書き、
120/// 古典的 xref テーブルとトレーラで締める。
121pub fn write_document(
122    version: &str,
123    objects: &std::collections::BTreeMap<crate::object::ObjectId, Object>,
124    trailer_extra: &Dictionary,
125) -> Result<Vec<u8>> {
126    let mut out = Vec::new();
127    // ヘッダ。2 行目のバイナリコメントは「バイナリファイルである」ことを
128    // 転送プログラムへ知らせる慣習（§7.5.2 Note）。
129    out.extend_from_slice(format!("%PDF-{version}\n").as_bytes());
130    out.extend_from_slice(b"%\xE2\xE3\xCF\xD3\n");
131
132    // 本体: オブジェクト番号順に書き出し、オフセットを記録
133    let mut offsets: Vec<(u32, u16, usize)> = Vec::with_capacity(objects.len());
134    for ((num, gen), obj) in objects {
135        offsets.push((*num, *gen, out.len()));
136        out.extend_from_slice(format!("{num} {gen} obj\n").as_bytes());
137        write_object(&mut out, obj);
138        out.extend_from_slice(b"\nendobj\n");
139    }
140
141    // xref テーブル（連続区間ごとにサブセクション化）
142    let xref_pos = out.len();
143    out.extend_from_slice(b"xref\n");
144    // オブジェクト 0 は常に空きリストの先頭
145    let mut entries: Vec<(u32, String)> = vec![(0, "0000000000 65535 f \n".into())];
146    for (num, gen, off) in &offsets {
147        entries.push((*num, format!("{off:010} {gen:05} n \n")));
148    }
149    entries.sort_by_key(|(n, _)| *n);
150    let mut i = 0;
151    while i < entries.len() {
152        let start = entries[i].0;
153        let mut j = i;
154        while j + 1 < entries.len() && entries[j + 1].0 == entries[j].0 + 1 {
155            j += 1;
156        }
157        out.extend_from_slice(format!("{start} {}\n", j - i + 1).as_bytes());
158        for (_, line) in &entries[i..=j] {
159            out.extend_from_slice(line.as_bytes());
160        }
161        i = j + 1;
162    }
163
164    // トレーラ
165    let max_num = objects.keys().map(|(n, _)| *n).max().unwrap_or(0);
166    let mut trailer = Dictionary::new();
167    trailer.set("Size", (max_num + 1) as i64);
168    for (k, v) in trailer_extra.iter() {
169        // 再生成するキーは引き継がない
170        if matches!(k, "Size" | "Prev" | "XRefStm" | "Type" | "W" | "Index" | "Length" | "Filter" | "DecodeParms" | "ID") {
171            continue;
172        }
173        trailer.set(k, v.clone());
174    }
175    // /ID: 本体バイト列のハッシュから 16 バイトの識別子を生成（§14.4）
176    let h1 = fnv1a64(&out, 0xcbf29ce484222325);
177    let h2 = fnv1a64(&out, 0x84222325cbf29ce4);
178    let mut id_bytes = Vec::with_capacity(16);
179    id_bytes.extend_from_slice(&h1.to_be_bytes());
180    id_bytes.extend_from_slice(&h2.to_be_bytes());
181    let id = Object::String(id_bytes, StringFormat::Hexadecimal);
182    trailer.set("ID", Object::Array(vec![id.clone(), id]));
183    out.extend_from_slice(b"trailer\n");
184    write_object(&mut out, &Object::Dictionary(trailer));
185    out.extend_from_slice(format!("\nstartxref\n{xref_pos}\n%%EOF\n").as_bytes());
186    Ok(out)
187}

189/// FNV-1a 64bit ハッシュ（/ID 生成用。暗号学的強度は不要）。
190fn fnv1a64(data: &[u8], seed: u64) -> u64 {
191    let mut h = seed;
192    for &b in data {
193        h ^= b as u64;
194        h = h.wrapping_mul(0x100000001b3);
195    }
196    h
197}

199#[cfg(test)]
200mod tests {
201    use super::*;
202    use crate::object::Stream;
203
204    fn ser(obj: &Object) -> String {
205        let mut v = Vec::new();
206        write_object(&mut v, obj);
207        String::from_utf8(v).unwrap()
208    }
209
210    #[test]
211    fn serialize_primitives() {
212        assert_eq!(ser(&Object::Null), "null");
213        assert_eq!(ser(&Object::Integer(-42)), "-42");
214        assert_eq!(ser(&Object::Real(3.5)), "3.5");
215        assert_eq!(ser(&Object::Real(2.0)), "2");
216        assert_eq!(ser(&Object::Reference((7, 0))), "7 0 R");
217        assert_eq!(ser(&Object::Name("A B".into())), "/A#20B");
218    }
219
220    #[test]
221    fn serialize_strings() {
222        assert_eq!(ser(&Object::string_literal("a(b)c\\")), "(a\\(b\\)c\\\\)");
223        assert_eq!(
224            ser(&Object::String(vec![0xFE, 0xFF], StringFormat::Hexadecimal)),
225            "<FEFF>"
226        );
227    }
228
229    #[test]
230    fn serialize_roundtrip_via_parser() {
231        let mut d = Dictionary::new();
232        d.set("Type", Object::Name("Test".into()));
233        d.set("Nums", Object::Array(vec![1.into(), 2.5.into(), Object::Reference((3, 0))]));
234        d.set("S", Object::string_literal("hello\nworld"));
235        let obj = Object::Dictionary(d);
236        let bytes = ser(&obj).into_bytes();
237        let parsed = crate::parser::Parser::new_at(&bytes, 0).parse_object().unwrap();
238        assert_eq!(parsed, obj);
239    }
240
241    #[test]
242    fn serialize_stream_roundtrip() {
243        let s = Stream::new(Dictionary::new(), b"DATA".to_vec());
244        let obj = Object::Stream(s);
245        let bytes = ser(&obj).into_bytes();
246        let parsed = crate::parser::Parser::new_at(&bytes, 0).parse_object().unwrap();
247        assert_eq!(parsed.as_stream().unwrap().data, b"DATA");
248    }
