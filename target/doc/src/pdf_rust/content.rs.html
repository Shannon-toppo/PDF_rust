1//! コンテントストリーム（ページ記述）の解析と生成。
2//!
3//! コンテントストリームは「オペランド列 + 演算子」の繰り返しで構成される
4//! （§7.8.2）。本モジュールはこれを [`Operation`] の列に分解する。
5//! インライン画像（`BI ... ID <バイナリ> EI`）はデータ部をスキップして
6//! 1 つの `BI` 演算として返す。
7
8use crate::error::{PdfError, Result};
9use crate::lexer::Token;
10use crate::object::Object;
11use crate::parser::Parser;
12
13/// コンテントストリーム中の演算 1 つ。
14#[derive(Debug, Clone, PartialEq)]
15pub struct Operation {
16    /// 演算子（`Tj` `re` `cm` など）。
17    pub operator: String,
18    /// オペランド（演算子の前に置かれた値）。
19    pub operands: Vec<Object>,
20}
21
22impl Operation {
23    /// 演算を作る補助関数。
24    pub fn new(operator: impl Into<String>, operands: Vec<Object>) -> Self {
25        Operation { operator: operator.into(), operands }
26    }
27}
28
29/// コンテントストリームを解析して演算列を返す。
30///
31/// 多少の構文エラー（不明な演算子など）は読み飛ばして継続する。
32pub fn parse_content(data: &[u8]) -> Result<Vec<Operation>> {
33    let mut parser = Parser::new_at(data, 0);
34    let mut ops = Vec::new();
35    let mut operands: Vec<Object> = Vec::new();
36    loop {
37        let token = match parser.lexer.next_token() {
38            Ok(t) => t,
39            Err(_) => continue, // 不正バイトは読み飛ばす
40        };
41        match token {
42            Token::Eof => break,
43            Token::Keyword(kw) => match kw.as_str() {
44                "true" => operands.push(Object::Boolean(true)),
45                "false" => operands.push(Object::Boolean(false)),
46                "null" => operands.push(Object::Null),
47                "BI" => {
48                    // インライン画像: 辞書部を読み、ID の後のバイナリを EI までスキップ
49                    let dict_ops = parse_inline_image(&mut parser)?;
50                    ops.push(Operation::new("BI", dict_ops));
51                    operands.clear();
52                }
53                _ => {
54                    ops.push(Operation::new(kw, std::mem::take(&mut operands)));
55                }
56            },
57            t => {
58                // 値はオペランドとして積む
59                match parser.parse_object_from(t) {
60                    Ok(o) => operands.push(o),
61                    Err(_) => operands.clear(),
62                }
63            }
64        }
65    }
66    Ok(ops)
67}

69/// `BI` 消費済みの状態からインライン画像を読み飛ばす。
70/// 戻り値は画像辞書のキーと値を交互に並べたオペランド列。
71fn parse_inline_image(parser: &mut Parser) -> Result<Vec<Object>> {
72    let mut kv = Vec::new();
73    loop {
74        match parser.lexer.next_token()? {
75            Token::Keyword(ref k) if k == "ID" => break,
76            Token::Eof => {
77                return Err(PdfError::Syntax {
78                    offset: parser.pos(),
79                    message: "unterminated inline image".into(),
80                })
81            }
82            t => kv.push(parser.parse_object_from(t)?),
83        }
84    }
85    // ID の直後 1 バイトの空白を挟んでバイナリデータが始まる
86    let data = parser.lexer.data;
87    let mut p = parser.pos();
88    if p < data.len() && crate::lexer::is_whitespace(data[p]) {
89        p += 1;
90    }
91    // 空白に挟まれた "EI" を探す
92    while p + 2 <= data.len() {
93        if &data[p..p + 2] == b"EI"
94            && (p == 0 || crate::lexer::is_whitespace(data[p - 1]))
95            && (p + 2 == data.len() || !crate::lexer::is_regular(data[p + 2]))
96        {
97            parser.lexer.pos = p + 2;
98            return Ok(kv);
99        }
100        p += 1;
101    }
102    Err(PdfError::Syntax { offset: parser.pos(), message: "missing EI for inline image".into() })
103}

105/// 演算列をコンテントストリームのバイト列に直列化する。
106pub fn write_content(ops: &[Operation]) -> Vec<u8> {
107    let mut out = Vec::new();
108    for op in ops {
109        for operand in &op.operands {
110            crate::writer::write_object(&mut out, operand);
111            out.push(b' ');
112        }
113        out.extend_from_slice(op.operator.as_bytes());
114        out.push(b'\n');
115    }
116    out
117}

119#[cfg(test)]
120mod tests {
121    use super::*;
122
123    #[test]
124    fn parses_simple_content() {
125        let src = b"BT /F1 12 Tf 72 700 Td (Hello) Tj ET 0 0 100 50 re S";
126        let ops = parse_content(src).unwrap();
127        let names: Vec<&str> = ops.iter().map(|o| o.operator.as_str()).collect();
128        assert_eq!(names, vec!["BT", "Tf", "Td", "Tj", "ET", "re", "S"]);
129        assert_eq!(ops[3].operands[0].as_string().unwrap(), b"Hello");
130        assert_eq!(ops[5].operands.len(), 4);
131    }
132
133    #[test]
134    fn skips_inline_image() {
135        let src = b"q BI /W 2 /H 2 /BPC 8 /CS /G ID \x00\x01\x02\x03 EI Q (x) Tj";
136        let ops = parse_content(src).unwrap();
137        let names: Vec<&str> = ops.iter().map(|o| o.operator.as_str()).collect();
138        assert_eq!(names, vec!["q", "BI", "Q", "Tj"]);
139    }
140
141    #[test]
142    fn content_roundtrip() {
143        let ops = vec![
144            Operation::new("BT", vec![]),
145            Operation::new("Tf", vec![Object::name("F1"), 12.into()]),
146            Operation::new("Td", vec![72.into(), 700.into()]),
147            Operation::new("Tj", vec![Object::string_literal("Hi")]),
148            Operation::new("ET", vec![]),
149        ];
150        let bytes = write_content(&ops);
151        assert_eq!(parse_content(&bytes).unwrap(), ops);
152    }
153}
