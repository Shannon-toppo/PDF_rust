1//! 相互参照テーブル（クロスリファレンス）の読み取り。
2//!
3//! 以下の 3 形式すべてに対応する:
4//! - 古典的な `xref` テーブル（§7.5.4）
5//! - クロスリファレンスストリーム（§7.5.8, PDF 1.5+）
6//! - ハイブリッド参照ファイル（`/XRefStm`）
7//!
8//! また、xref が壊れている場合はファイル全体を走査して
9//! `n g obj` を拾い集める再構築（[`reconstruct`]）を行う。
10

18/// 1 オブジェクト分の xref エントリ。
19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum XrefEntry {
21    /// 空きエントリ（type 0 / `f`）
22    Free,
23    /// ファイル内に非圧縮で存在（type 1 / `n`）。`offset` はファイル先頭からの位置。
24    InFile { offset: usize, gen: u16 },
25    /// オブジェクトストリーム内に圧縮格納（type 2）。
26    InStream { stream_num: u32, index: u32 },
27}
28
29/// 読み取った xref 全体。
30#[derive(Debug, Default)]
31pub struct Xref {
32    /// オブジェクト番号 → エントリ。更新が重なる場合は最新のものだけを保持。
33    pub entries: HashMap<u32, XrefEntry>,
34    /// マージ済みトレーラ辞書（新しい更新のキーを優先）。
35    pub trailer: Dictionary,
36}
37
38impl Xref {
39    /// `startxref` を探して xref チェーン全体を読み取る。
40    pub fn load(data: &[u8]) -> Result<Xref> {
41        let start = find_startxref(data)?;
42        let mut xref = Xref::default();
43        let mut visited = Vec::new();
44        let mut next = Some(start);
45        while let Some(offset) = next {
46            if visited.contains(&offset) {
47                break; // ループ防止
48            }
49            visited.push(offset);
50            if offset >= data.len() {
51                return Err(PdfError::BrokenXref(format!("xref offset {offset} out of range")));
52            }
53            let trailer = parse_section(data, offset, &mut xref)?;
54            // ハイブリッド参照: 古典テーブルの trailer が /XRefStm を持つ場合、
55            // そのストリームのエントリは古典テーブルより優先度が低い扱いで読む
56            // （既存エントリを上書きしない merge 規則で自然に処理される）。
57            if let Some(Object::Integer(s)) = trailer.get("XRefStm") {
58                let s = *s as usize;
59                if !visited.contains(&s) && s < data.len() {
60                    visited.push(s);
61                    let _ = parse_section(data, s, &mut xref);
62                }
63            }
64            next = match trailer.get("Prev") {
65                Some(Object::Integer(p)) if *p >= 0 => Some(*p as usize),
66                _ => None,
67            };
68            merge_trailer(&mut xref.trailer, trailer);
69        }
70        if xref.entries.is_empty() {
71            return Err(PdfError::BrokenXref("no xref entries found".into()));
72        }
73        Ok(xref)
74    }
75
76    /// エントリを追加する。既にあるオブジェクト番号は上書きしない
77    /// （新しい更新から順に読むため、先に見つかったものが最新）。
78    fn insert(&mut self, num: u32, entry: XrefEntry) {
79        self.entries.entry(num).or_insert(entry);
80    }
81}
82
83/// トレーラのマージ。先に読んだ（=新しい）キーを優先する。
84fn merge_trailer(dst: &mut Dictionary, src: Dictionary) {
85    for (k, v) in src.iter() {
86        if !dst.contains_key(k) {
87            dst.set(k, v.clone());
88        }
89    }
90}

92/// ファイル末尾から `startxref` を探してオフセットを返す。
93pub fn find_startxref(data: &[u8]) -> Result<usize> {
94    let tail_len = data.len().min(2048);
95    let tail_start = data.len() - tail_len;
96    let tail = &data[tail_start..];
97    let pos = find_last(tail, b"startxref")
98        .ok_or_else(|| PdfError::BrokenXref("'startxref' not found".into()))?;
99    let mut lexer = Lexer::new_at(data, tail_start + pos + b"startxref".len());
100    match lexer.next_token() {
101        Ok(Token::Integer(n)) if n >= 0 => Ok(n as usize),
102        _ => Err(PdfError::BrokenXref("invalid startxref offset".into())),
103    }
104}
105
106fn find_last(haystack: &[u8], needle: &[u8]) -> Option<usize> {
107    if needle.is_empty() || haystack.len() < needle.len() {
108        return None;
109    }
110    (0..=haystack.len() - needle.len()).rev().find(|&i| &haystack[i..i + needle.len()] == needle)
111}

113/// `offset` にある xref セクション 1 つ（古典 or ストリーム）を読み、
114/// そのセクションのトレーラ辞書を返す。
115fn parse_section(data: &[u8], offset: usize, xref: &mut Xref) -> Result<Dictionary> {
116    let mut lexer = Lexer::new_at(data, offset);
117    let save = lexer.pos;
118    match lexer.next_token() {
119        Ok(Token::Keyword(ref k)) if k == "xref" => parse_classic_table(data, lexer.pos, xref),
120        _ => {
121            // クロスリファレンスストリーム（間接オブジェクト）
122            let mut parser = Parser::new_at(data, save);
123            let (_, obj) = parser.parse_indirect_object()?;
124            let stream = obj.as_stream()?;
125            parse_xref_stream(stream, xref)?;
126            Ok(stream.dict.clone())
127        }
128    }
129}

131/// 古典的な xref テーブルを読む。`xref` キーワードは消費済み。
132fn parse_classic_table(data: &[u8], pos: usize, xref: &mut Xref) -> Result<Dictionary> {
133    let mut lexer = Lexer::new_at(data, pos);
134    loop {
135        let save = lexer.pos;
136        match lexer.next_token() {
137            // サブセクション: `開始番号 個数`
138            Ok(Token::Integer(start)) => {
139                let count = match lexer.next_token() {
140                    Ok(Token::Integer(c)) if c >= 0 => c as u32,
141                    t => {
142                        return Err(PdfError::BrokenXref(format!(
143                            "invalid xref subsection header: {t:?}"
144                        )))
145                    }
146                };
147                if start < 0 {
148                    return Err(PdfError::BrokenXref("negative subsection start".into()));
149                }
150                // 各エントリは「10 桁オフセット 5 桁世代 n/f」の固定 20 バイトだが、
151                // 改行の乱れに耐えるためトークン単位で読む。
152                for i in 0..count {
153                    let num = start as u32 + i;
154                    let f1 = match lexer.next_token() {
155                        Ok(Token::Integer(v)) if v >= 0 => v as usize,
156                        t => {
157                            return Err(PdfError::BrokenXref(format!("bad xref entry: {t:?}")))
158                        }
159                    };
160                    let f2 = match lexer.next_token() {
161                        Ok(Token::Integer(v)) if v >= 0 => v as u16,
162                        Ok(Token::Integer(_)) => 0,
163                        t => {
164                            return Err(PdfError::BrokenXref(format!("bad xref entry: {t:?}")))
165                        }
166                    };
167                    match lexer.next_token() {
168                        Ok(Token::Keyword(ref k)) if k == "n" => {
169                            xref.insert(num, XrefEntry::InFile { offset: f1, gen: f2 });
170                        }
171                        Ok(Token::Keyword(ref k)) if k == "f" => {
172                            xref.insert(num, XrefEntry::Free);
173                        }
174                        t => {
175                            return Err(PdfError::BrokenXref(format!(
176                                "bad xref entry type: {t:?}"
177                            )))
178                        }
179                    }
180                }
181            }
182            // trailer
183            Ok(Token::Keyword(ref k)) if k == "trailer" => {
184                let mut parser = Parser::new_at(data, lexer.pos);
185                let obj = parser.parse_object()?;
186                return Ok(obj.as_dict()?.clone());
187            }
188            t => {
189                let _ = save;
190                return Err(PdfError::BrokenXref(format!(
191                    "unexpected token in xref table: {t:?}"
192                )));
193            }
194        }
195    }
196}

198/// クロスリファレンスストリーム（§7.5.8）を読む。
199fn parse_xref_stream(stream: &crate::object::Stream, xref: &mut Xref) -> Result<()> {
200    let dict = &stream.dict;
201    let decoded = crate::filters::decode_stream(dict, &stream.data, None)?;
202
203    // /W: 各フィールドのバイト幅
204    let w = dict.require("W")?.as_array()?;
205    let widths: Vec<usize> = w
206        .iter()
207        .map(|o| o.as_int().map(|v| v as usize))
208        .collect::<Result<_>>()
209        .map_err(|_| PdfError::BrokenXref("invalid /W array".into()))?;
210    if widths.len() < 3 {
211        return Err(PdfError::BrokenXref("/W must have 3 elements".into()));
212    }
213    let row_len: usize = widths.iter().sum();
214    if row_len == 0 {
215        return Err(PdfError::BrokenXref("zero-width xref rows".into()));
216    }
217
218    // /Index: [開始 個数 ...]（省略時は [0 Size]）
219    let size = dict.require("Size")?.as_int()?;
220    let index: Vec<i64> = match dict.get("Index") {
221        Some(Object::Array(a)) => {
222            a.iter().map(|o| o.as_int()).collect::<Result<_>>()?
223        }
224        _ => vec![0, size],
225    };
226
227    let read_field = |bytes: &[u8]| -> u64 {
228        bytes.iter().fold(0u64, |acc, &b| (acc << 8) | b as u64)
229    };
230
231    let mut row_iter = decoded.chunks_exact(row_len);
232    for pair in index.chunks(2) {
233        let (start, count) = (pair[0], *pair.get(1).unwrap_or(&0));
234        for i in 0..count {
235            let row = match row_iter.next() {
236                Some(r) => r,
237                None => return Err(PdfError::BrokenXref("xref stream too short".into())),
238            };
239            let mut p = 0;
240            // type フィールド幅 0 のときの既定値は 1（§7.5.8.3）
241            let ftype = if widths[0] == 0 { 1 } else { read_field(&row[..widths[0]]) };
242            p += widths[0];
243            let f2 = read_field(&row[p..p + widths[1]]);
244            p += widths[1];
245            let f3 = read_field(&row[p..p + widths[2]]);
246            let num = (start + i) as u32;
247            match ftype {
248                0 => xref.insert(num, XrefEntry::Free),
249                1 => xref.insert(num, XrefEntry::InFile { offset: f2 as usize, gen: f3 as u16 }),
250                2 => xref.insert(
251                    num,
252                    XrefEntry::InStream { stream_num: f2 as u32, index: f3 as u32 },
253                ),
254                _ => {} // 未知タイプは無視（将来拡張、§7.5.8.3）
255            }
256        }
257    }
258    Ok(())
259}

261/// 壊れた PDF のための xref 再構築。
262///
263/// ファイル全体を走査して `n g obj` の出現位置を収集する。
264/// 同じオブジェクト番号が複数回現れた場合は最後（=最新）を採用する。
265pub fn reconstruct(data: &[u8]) -> Result<Xref> {
266    let mut xref = Xref::default();
267    let mut i = 0usize;
268    while i < data.len() {
269        // "obj" を探す
270        if data[i..].starts_with(b"obj")
271            && (i + 3 >= data.len() || !crate::lexer::is_regular(data[i + 3]))
272        {
273            // 後ろ向きに `n g` を読む
274            if let Some((num, gen, start)) = scan_back_obj_header(data, i) {
275                xref.entries.insert(num, XrefEntry::InFile { offset: start, gen });
276            }
277        }
278        i += 1;
279    }
280    if xref.entries.is_empty() {
281        return Err(PdfError::BrokenXref("reconstruction found no objects".into()));
282    }
283    // trailer 辞書も探す（最後のものを採用）
284    if let Some(tpos) = find_last(data, b"trailer") {
285        let mut parser = Parser::new_at(data, tpos + b"trailer".len());
286        if let Ok(obj) = parser.parse_object() {
287            if let Ok(d) = obj.as_dict() {
288                xref.trailer = d.clone();
289            }
290        }
291    }
292    Ok(xref)
293}

295/// `obj` キーワードの位置から後ろ向きに「番号 世代」を読み取る。
296/// 戻り値は `(番号, 世代, オブジェクト開始位置)`。
297fn scan_back_obj_header(data: &[u8], obj_pos: usize) -> Option<(u32, u16, usize)> {
298    let mut p = obj_pos;
299    // 空白を戻る
300    let skip_ws_back = |p: &mut usize| {
301        while *p > 0 && crate::lexer::is_whitespace(data[*p - 1]) {
302            *p -= 1;
303        }
304    };
305    let read_int_back = |p: &mut usize| -> Option<(u64, usize)> {
306        let end = *p;
307        while *p > 0 && data[*p - 1].is_ascii_digit() {
308            *p -= 1;
309        }
310        if *p == end {
311            return None;
312        }
313        let s = std::str::from_utf8(&data[*p..end]).ok()?;
314        Some((s.parse().ok()?, *p))
315    };
316    skip_ws_back(&mut p);
317    let (gen, _) = read_int_back(&mut p)?;
318    skip_ws_back(&mut p);
319    let (num, start) = read_int_back(&mut p)?;
320    if gen > u16::MAX as u64 || num > u32::MAX as u64 {
321        return None;
322    }
323    Some((num as u32, gen as u16, start))
324}

326#[cfg(test)]
327mod tests {

329
330    #[test]
331    fn classic_table() {
332        let pdf = b"xref\n0 3\n0000000000 65535 f \n0000000017 00000 n \n0000000081 00000 n \ntrailer\n<< /Size 3 /Root 1 0 R >>\nstartxref\n0\n%%EOF";
333        let mut xref = Xref::default();
334        let trailer = parse_section(pdf, 0, &mut xref).unwrap();
335        assert_eq!(xref.entries[&0], XrefEntry::Free);
336        assert_eq!(xref.entries[&1], XrefEntry::InFile { offset: 17, gen: 0 });
337        assert_eq!(trailer.get("Size").unwrap().as_int().unwrap(), 3);
338    }
339
340    #[test]
341    fn reconstruct_finds_objects() {
342        let pdf = b"%PDF-1.4\n1 0 obj << /A 1 >> endobj\n2 0 obj << /B 2 >> endobj\n1 0 obj << /A 9 >> endobj\n";
343        let xref = reconstruct(pdf).unwrap();
344        // 1 0 obj は 2 回現れる: 最後の出現（offset 62）が勝つ
345        match xref.entries[&1] {
346            XrefEntry::InFile { offset, .. } => assert!(offset > 35),
347            _ => panic!(),
348        }
349        assert!(matches!(xref.entries[&2], XrefEntry::InFile { .. }));
350    }
