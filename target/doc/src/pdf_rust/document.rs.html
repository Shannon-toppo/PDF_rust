1//! PDF ドキュメントの読み込み・編集・保存。
2//!
3//! [`Document`] が本ライブラリの中心 API。ファイル全体をメモリに読み込み、
4//! すべての間接オブジェクトを `(番号, 世代)` → [`Object`] のマップとして
5//! 保持する。編集後は [`Document::save`] で完全書き直し（非増分更新）の
6//! PDF を出力する。
7
8use std::collections::{BTreeMap, HashSet};
9use std::path::Path;
10
11use crate::content::{write_content, Operation};
12use crate::error::{PdfError, Result};
13use crate::font::StandardFont;
14use crate::object::{Dictionary, Object, ObjectId, Stream};
15use crate::parser::Parser;
16use crate::xref::{self, Xref, XrefEntry};

18static NULL_OBJ: Object = Object::Null;

20/// メモリ上の PDF ドキュメント。
21#[derive(Debug)]
22pub struct Document {
23    /// PDF バージョン（`"1.7"` など）。
24    pub version: String,
25    /// すべての間接オブジェクト。
26    pub objects: BTreeMap<ObjectId, Object>,
27    /// トレーラ辞書（`/Root` `/Info` などを保持。`/Size` 等は保存時に再生成）。
28    pub trailer: Dictionary,
29    next_id: u32,
30}
31

33// 生成・読み込み
34// ---------------------------------------------------------------------------

36impl Document {
37    /// 空のドキュメント（カタログ + 空のページツリー）を作る。
38    pub fn new() -> Document {
39        let mut doc = Document {
40            version: "1.7".into(),
41            objects: BTreeMap::new(),
42            trailer: Dictionary::new(),
43            next_id: 1,
44        };
45        let mut pages = Dictionary::new();
46        pages.set("Type", Object::name("Pages"));
47        pages.set("Kids", Object::Array(vec![]));
48        pages.set("Count", 0);
49        let pages_id = doc.add_object(pages);
50
51        let mut catalog = Dictionary::new();
52        catalog.set("Type", Object::name("Catalog"));
53        catalog.set("Pages", Object::Reference(pages_id));
54        let catalog_id = doc.add_object(catalog);
55
56        doc.trailer.set("Root", Object::Reference(catalog_id));
57        doc
58    }
59
60    /// ファイルから読み込む。
61    pub fn load(path: impl AsRef<Path>) -> Result<Document> {
62        let data = std::fs::read(path)?;
63        Self::from_bytes(&data)
64    }
65
66    /// バイト列から読み込む。
67    ///
68    /// xref が壊れている場合は全ファイル走査による再構築を試みる。
69    pub fn from_bytes(data: &[u8]) -> Result<Document> {
70        let version = parse_header_version(data)?;
71        let primary = Xref::load(data).and_then(|x| Self::build(data, x, version.clone()));
72        match primary {
73            Ok(doc) => Ok(doc),
74            Err(PdfError::EncryptionNotSupported) => Err(PdfError::EncryptionNotSupported),
75            Err(_) => {
76                let x = xref::reconstruct(data)?;
77                Self::build(data, x, version)
78            }
79        }
80    }
81
82    /// xref に従ってすべてのオブジェクトを読み込む。
83    fn build(data: &[u8], xref: Xref, version: String) -> Result<Document> {
84        if xref.trailer.get("Encrypt").is_some() {
85            return Err(PdfError::EncryptionNotSupported);
86        }
87
88        // /Length 間接参照の解決用: xref から番号を引いて単純オブジェクトを読む
89        let length_resolver = |id: ObjectId| -> Option<i64> {
90            match xref.entries.get(&id.0)? {
91                XrefEntry::InFile { offset, .. } => {
92                    let mut p = Parser::new_at(data, *offset);
93                    let (_, obj) = p.parse_indirect_object().ok()?;
94                    obj.as_int().ok()
95                }
96                _ => None,
97            }
98        };
99
100        // 第 1 段階: ファイル直置きのオブジェクト
101        let mut objects: BTreeMap<ObjectId, Object> = BTreeMap::new();
102        for (&num, &entry) in &xref.entries {
103            if let XrefEntry::InFile { offset, .. } = entry {
104                if offset >= data.len() {
105                    continue; // 範囲外エントリは無視（耐性優先）
106                }
107                let mut p = Parser::new_at(data, offset);
108                p.length_resolver = Some(&length_resolver);
109                match p.parse_indirect_object() {
110                    Ok(((n, g), obj)) if n == num => {
111                        objects.insert((n, g), obj);
112                    }
113                    // 番号不一致・解析失敗のエントリは無視（後段の再構築に任せる）
114                    _ => {}
115                }
116            }
117        }
118
119        // 第 2 段階: オブジェクトストリーム内の圧縮オブジェクト
120        let stream_nums: HashSet<u32> = xref
121            .entries
122            .values()
123            .filter_map(|e| match e {
124                XrefEntry::InStream { stream_num, .. } => Some(*stream_num),
125                _ => None,
126            })
127            .collect();
128        let mut extracted: Vec<(ObjectId, Object)> = Vec::new();
129        for snum in stream_nums {
130            let stream = match objects.get(&(snum, 0)) {
131                Some(Object::Stream(s)) => s.clone(),
132                _ => continue,
133            };
134            let resolve = |o: &Object| -> Object {
135                match o {
136                    Object::Reference(id) => objects.get(id).cloned().unwrap_or(Object::Null),
137                    other => other.clone(),
138                }
139            };
140            let decoded =
141                match crate::filters::decode_stream(&stream.dict, &stream.data, Some(&resolve)) {
142                    Ok(d) => d,
143                    Err(_) => continue,
144                };
145            let n = match stream.dict.get("N").map(&resolve) {
146                Some(o) => o.as_int().unwrap_or(0) as usize,
147                None => 0,
148            };
149            let first = match stream.dict.get("First").map(resolve) {
150                Some(o) => o.as_int().unwrap_or(0) as usize,
151                None => 0,
152            };
153            // ヘッダ部: 「オブジェクト番号 オフセット」の対が N 個
154            let mut header = Parser::new_at(&decoded, 0);
155            let mut pairs = Vec::with_capacity(n);
156            for _ in 0..n {
157                let onum = match header.lexer.next_token() {
158                    Ok(crate::lexer::Token::Integer(v)) if v >= 0 => v as u32,
159                    _ => break,
160                };
161                let ooff = match header.lexer.next_token() {
162                    Ok(crate::lexer::Token::Integer(v)) if v >= 0 => v as usize,
163                    _ => break,
164                };
165                pairs.push((onum, ooff));
166            }
167            for (onum, ooff) in pairs {
168                // この番号が確かにこのストリーム内を指しているときだけ採用
169                match xref.entries.get(&onum) {
170                    Some(XrefEntry::InStream { stream_num, .. }) if *stream_num == snum => {}
171                    _ => continue,
172                }
173                if first + ooff >= decoded.len() {
174                    continue;
175                }
176                if let Ok(obj) = Parser::new_at(&decoded, first + ooff).parse_object() {
177                    extracted.push(((onum, 0), obj));
178                }
179            }
180        }
181        for (id, obj) in extracted {
182            objects.insert(id, obj);
183        }
184
185        // ObjStm / XRef ストリームは保存時に再生成されるため取り除く
186        objects.retain(|_, obj| {
187            if let Object::Stream(s) = obj {
188                if let Some(Object::Name(t)) = s.dict.get("Type") {
189                    return t != "ObjStm" && t != "XRef";
190                }
191            }
192            true
193        });
194
195        if objects.is_empty() {
196            return Err(PdfError::BrokenXref("no objects could be loaded".into()));
197        }
198
199        let mut trailer = xref.trailer;
200        for k in ["Prev", "XRefStm", "Size", "Type", "W", "Index", "Length", "Filter", "DecodeParms"] {
201            trailer.remove(k);
202        }
203        let next_id = objects.keys().map(|(n, _)| *n).max().unwrap_or(0) + 1;
204        let doc = Document { version, objects, trailer, next_id };
205        // 最低限の検証: カタログが引けること
206        doc.catalog()?;
207        Ok(doc)
208    }
209}

221impl Document {
222    /// 新しい間接オブジェクトを登録し、その ID を返す。
223    pub fn add_object(&mut self, obj: impl Into<Object>) -> ObjectId {
224        let id = (self.next_id, 0);
225        self.next_id += 1;
226        self.objects.insert(id, obj.into());
227        id
228    }
229
230    /// ID でオブジェクトを取得する。世代番号が一致しない場合は
231    /// 同じ番号の別世代も探す（壊れた PDF への耐性）。
232    pub fn get_object(&self, id: ObjectId) -> Result<&Object> {
233        if let Some(o) = self.objects.get(&id) {
234            return Ok(o);
235        }
236        self.objects
237            .range((id.0, 0)..=(id.0, u16::MAX))
238            .next()
239            .map(|(_, o)| o)
240            .ok_or(PdfError::MissingObject(id.0, id.1))
241    }
242
243    /// ID でオブジェクトを可変取得する。
244    pub fn get_object_mut(&mut self, id: ObjectId) -> Result<&mut Object> {
245        let key = if self.objects.contains_key(&id) {
246            id
247        } else {
248            *self
249                .objects
250                .range((id.0, 0)..=(id.0, u16::MAX))
251                .next()
252                .map(|(k, _)| k)
253                .ok_or(PdfError::MissingObject(id.0, id.1))?
254        };
255        Ok(self.objects.get_mut(&key).unwrap())
256    }
257
258    /// 間接参照を辿って実体を返す（参照でなければそのまま返す）。
259    /// 参照切れは `Object::Null` になる。
260    pub fn resolve<'a>(&'a self, mut obj: &'a Object) -> &'a Object {
261        for _ in 0..64 {
262            match obj {
263                Object::Reference(id) => match self.get_object(*id) {
264                    Ok(o) => obj = o,
265                    Err(_) => return &NULL_OBJ,
266                },
267                other => return other,
268            }
269        }
270        &NULL_OBJ
271    }
272
273    /// 辞書 `dict` のキー `key` を取得し、間接参照も解決して返す。
274    pub fn dict_get<'a>(&'a self, dict: &'a Dictionary, key: &str) -> Option<&'a Object> {
275        dict.get(key).map(|o| self.resolve(o)).filter(|o| !matches!(o, Object::Null))
276    }
277
278    /// カタログ（ルート辞書）を返す。
279    pub fn catalog(&self) -> Result<&Dictionary> {
280        let root = self.trailer.require("Root")?;
281        self.resolve(root).as_dict()
282    }
283
284    /// ストリームの中身を `/Filter` を適用して伸長する。
285    /// `/Length` や `/Filter` 内の間接参照も解決される。
286    pub fn get_stream_data(&self, stream: &Stream) -> Result<Vec<u8>> {
287        let resolve = |o: &Object| self.resolve(o).clone();
288        crate::filters::decode_stream(&stream.dict, &stream.data, Some(&resolve))
289    }
290}

296impl Document {
297    /// 全ページの ID を文書順で返す。
298    pub fn pages(&self) -> Vec<ObjectId> {
299        let mut out = Vec::new();
300        let pages_ref = self
301            .catalog()
302            .ok()
303            .and_then(|c| c.get("Pages").cloned())
304            .and_then(|o| o.as_reference().ok());
305        if let Some(root) = pages_ref {
306            let mut visited = HashSet::new();
307            self.walk_page_tree(root, &mut out, &mut visited, 0);
308        }
309        out
310    }
311
312    fn walk_page_tree(
313        &self,
314        node_id: ObjectId,
315        out: &mut Vec<ObjectId>,
316        visited: &mut HashSet<u32>,
317        depth: usize,
318    ) {
319        if depth > 64 || !visited.insert(node_id.0) {
320            return; // 循環・過深防止
321        }
322        let dict = match self.get_object(node_id).ok().and_then(|o| o.as_dict().ok()) {
323            Some(d) => d,
324            None => return,
325        };
326        let node_type = dict.get("Type").and_then(|o| o.as_name().ok());
327        let is_pages = node_type == Some("Pages") || (node_type.is_none() && dict.contains_key("Kids"));
328        if is_pages {
329            if let Some(Object::Array(kids)) = self.dict_get(dict, "Kids") {
330                let kid_ids: Vec<ObjectId> =
331                    kids.iter().filter_map(|k| k.as_reference().ok()).collect();
332                for kid in kid_ids {
333                    self.walk_page_tree(kid, out, visited, depth + 1);
334                }
335            }
336        } else {
337            out.push(node_id);
338        }
339    }
340
341    /// ページ数。
342    pub fn page_count(&self) -> usize {
343        self.pages().len()
344    }
345
346    /// `index`（0 始まり）番目のページ ID。
347    pub fn page_id(&self, index: usize) -> Result<ObjectId> {
348        let pages = self.pages();
349        let count = pages.len();
350        pages.into_iter().nth(index).ok_or(PdfError::PageOutOfRange { index, count })
351    }
352
353    /// ページの継承属性（`/Resources` `/MediaBox` `/Rotate` など）を、
354    /// ページ自身 → 祖先の順で探して返す。
355    pub fn page_attr<'a>(&'a self, page_id: ObjectId, key: &str) -> Option<&'a Object> {
356        let mut current = page_id;
357        for _ in 0..64 {
358            let dict = self.get_object(current).ok()?.as_dict().ok()?;
359            if let Some(v) = self.dict_get(dict, key) {
360                return Some(v);
361            }
362            current = dict.get("Parent")?.as_reference().ok()?;
363        }
364        None
365    }
366
367    /// ページの MediaBox `[x0 y0 x1 y1]` を返す（継承解決込み。既定は Letter）。
368    pub fn page_media_box(&self, page_id: ObjectId) -> [f64; 4] {
369        if let Some(Object::Array(a)) = self.page_attr(page_id, "MediaBox") {
370            if a.len() == 4 {
371                let mut v = [0.0; 4];
372                for (i, o) in a.iter().enumerate() {
373                    v[i] = self.resolve(o).as_number().unwrap_or(0.0);
374                }
375                return v;
376            }
377        }
378        [0.0, 0.0, 612.0, 792.0]
379    }
380
381    /// ページのコンテントストリームをすべて連結して伸長したバイト列を返す。
382    pub fn page_content_bytes(&self, page_id: ObjectId) -> Result<Vec<u8>> {
383        let dict = self.get_object(page_id)?.as_dict()?;
384        let contents = match self.dict_get(dict, "Contents") {
385            Some(c) => c,
386            None => return Ok(Vec::new()),
387        };
388        let mut out = Vec::new();
389        match contents {
390            Object::Stream(s) => out = self.get_stream_data(s)?,
391            Object::Array(items) => {
392                for item in items {
393                    if let Object::Stream(s) = self.resolve(item) {
394                        out.extend_from_slice(&self.get_stream_data(s)?);
395                        out.push(b'\n'); // ストリーム境界はトークン境界（§7.8.2）
396                    }
397                }
398            }
399            _ => {}
400        }
401        Ok(out)
402    }
403
404    /// ページの実効 `/Resources` 辞書（継承解決込み）を返す。なければ空辞書。
405    pub fn page_resources(&self, page_id: ObjectId) -> Dictionary {
406        match self.page_attr(page_id, "Resources") {
407            Some(Object::Dictionary(d)) => d.clone(),
408            _ => Dictionary::new(),
409        }
410    }
411
412    /// ページのテキストを抽出する（`index` は 0 始まり）。
413    pub fn extract_text(&self, index: usize) -> Result<String> {
414        let page_id = self.page_id(index)?;
415        crate::text::extract_page_text(self, page_id)
416    }

424    /// 新しい空ページを末尾に追加する。`width` x `height` はポイント単位
425    /// （A4 = 595 x 842, Letter = 612 x 792）。追加したページの ID を返す。
426    pub fn add_page(&mut self, width: f64, height: f64) -> Result<ObjectId> {
427        let pages_id = self
428            .catalog()?
429            .require("Pages")?
430            .as_reference()?;
431        let mut page = Dictionary::new();
432        page.set("Type", Object::name("Page"));
433        page.set("Parent", Object::Reference(pages_id));
434        page.set(
435            "MediaBox",
436            Object::Array(vec![0.into(), 0.into(), width.into(), height.into()]),
437        );
438        page.set("Resources", Dictionary::new());
439        let page_id = self.add_object(page);
440
441        let pages_node = self.get_object_mut(pages_id)?.as_dict_mut()?;
442        match pages_node.get_mut("Kids") {
443            Some(Object::Array(kids)) => kids.push(Object::Reference(page_id)),
444            _ => pages_node.set("Kids", Object::Array(vec![Object::Reference(page_id)])),
445        }
446        let count = pages_node.get("Count").and_then(|o| o.as_int().ok()).unwrap_or(0);
447        pages_node.set("Count", count + 1);
448        Ok(page_id)
449    }
450
451    /// ページを削除する（`index` は 0 始まり）。
452    pub fn remove_page(&mut self, index: usize) -> Result<()> {
453        let page_id = self.page_id(index)?;
454        let parent_id = self
455            .get_object(page_id)?
456            .as_dict()?
457            .get("Parent")
458            .and_then(|o| o.as_reference().ok())
459            .ok_or(PdfError::Invalid("page has no /Parent".into()))?;
460
461        // 親の Kids から取り除く
462        let parent = self.get_object_mut(parent_id)?.as_dict_mut()?;
463        if let Some(Object::Array(kids)) = parent.get_mut("Kids") {
464            kids.retain(|k| k.as_reference().ok() != Some(page_id));
465        }
466        // 祖先すべての Count を 1 減らす
467        let mut current = Some(parent_id);
468        let mut guard = 0;
469        while let Some(node_id) = current {
470            guard += 1;
471            if guard > 64 {
472                break;
473            }
474            let node = self.get_object_mut(node_id)?.as_dict_mut()?;
475            let count = node.get("Count").and_then(|o| o.as_int().ok()).unwrap_or(1);
476            node.set("Count", (count - 1).max(0));
477            current = node.get("Parent").and_then(|o| o.as_reference().ok());
478        }
479        self.objects.remove(&page_id);
480        Ok(())
481    }

483    /// ページを回転する。`degrees` は 90 の倍数（時計回り）。既存の回転に加算される。
484    pub fn rotate_page(&mut self, index: usize, degrees: i64) -> Result<()> {
485        if degrees % 90 != 0 {
486            return Err(PdfError::Invalid("rotation must be a multiple of 90".into()));
487        }
488        let page_id = self.page_id(index)?;
489        let current = self
490            .page_attr(page_id, "Rotate")
491            .and_then(|o| o.as_int().ok())
492            .unwrap_or(0);
493        let dict = self.get_object_mut(page_id)?.as_dict_mut()?;
494        dict.set("Rotate", (current + degrees).rem_euclid(360));
495        Ok(())
496    }

498    /// ページ末尾にコンテントストリームを 1 本追加する。
499    ///
500    /// 既存コンテンツのグラフィックス状態の影響を避けたい場合は
501    /// 呼び出し側で `q`/`Q` や `BT`/`ET` で囲むこと（本ライブラリの
502    /// 高レベル API は囲み済みの演算列を生成する）。
503    pub fn append_content(&mut self, index: usize, ops: &[Operation]) -> Result<()> {
504        let bytes = write_content(ops);
505        self.append_content_bytes(index, bytes)
506    }

508    /// ページ末尾に生のコンテントストリームを追加する（低レベル API）。
509    pub fn append_content_bytes(&mut self, index: usize, bytes: Vec<u8>) -> Result<()> {
510        let page_id = self.page_id(index)?;
511        let stream_id = self.add_object(Object::Stream(Stream::new(Dictionary::new(), bytes)));
512
513        // 既存の /Contents を配列に正規化して追記
514        let existing = {
515            let dict = self.get_object(page_id)?.as_dict()?;
516            dict.get("Contents").cloned()
517        };
518        let new_contents = match existing {
519            None | Some(Object::Null) => Object::Array(vec![Object::Reference(stream_id)]),
520            Some(Object::Array(mut a)) => {
521                a.push(Object::Reference(stream_id));
522                Object::Array(a)
523            }
524            Some(Object::Reference(rid)) => {
525                // 参照先が配列か単一ストリームかで分岐
526                match self.get_object(rid) {
527                    Ok(Object::Array(a)) => {
528                        let mut a = a.clone();
529                        a.push(Object::Reference(stream_id));
530                        Object::Array(a)
531                    }
532                    _ => Object::Array(vec![
533                        Object::Reference(rid),
534                        Object::Reference(stream_id),
535                    ]),
536                }
537            }
538            Some(other) => other, // 不正な形は触らない
539        };
540        let dict = self.get_object_mut(page_id)?.as_dict_mut()?;
541        dict.set("Contents", new_contents);
542        Ok(())
543    }

545    /// ページに標準 14 フォントのリソースを登録し、リソース名（`F1` など）を返す。
546    /// 同じフォントが登録済みならその名前を返す。
547    pub fn ensure_standard_font(
548        &mut self,
549        index: usize,
550        font: StandardFont,
551    ) -> Result<String> {
552        let page_id = self.page_id(index)?;
553        // 実効リソースを直接辞書としてページに持たせる（継承・間接参照を解消）
554        let mut resources = self.page_resources(page_id);
555        let mut fonts = match resources.get("Font").map(|o| self.resolve(o)) {
556            Some(Object::Dictionary(d)) => d.clone(),
557            _ => Dictionary::new(),
558        };
559
560        // 既存エントリの再利用
561        for (name, value) in fonts.iter() {
562            if let Ok(fd) = self.resolve(value).as_dict() {
563                if fd.get("BaseFont").and_then(|o| o.as_name().ok()) == Some(font.base_font()) {
564                    return Ok(name.to_string());
565                }
566            }
567        }
568
569        // 新規登録
570        let mut fdict = Dictionary::new();
571        fdict.set("Type", Object::name("Font"));
572        fdict.set("Subtype", Object::name("Type1"));
573        fdict.set("BaseFont", Object::name(font.base_font()));
574        fdict.set("Encoding", Object::name("WinAnsiEncoding"));
575        let fid = self.add_object(fdict);
576
577        let mut n = 1;
578        let name = loop {
579            let candidate = format!("F{n}");
580            if !fonts.contains_key(&candidate) {
581                break candidate;
582            }
583            n += 1;
584        };
585        fonts.set(name.clone(), Object::Reference(fid));
586        resources.set("Font", fonts);
587        let dict = self.get_object_mut(page_id)?.as_dict_mut()?;
588        dict.set("Resources", resources);
589        Ok(name)
590    }
591
592    /// ページにテキストを描画する。複数行（`\n` 区切り）に対応。
593    ///
594    /// 座標系は PDF 標準（原点は左下、y 軸は上向き、単位はポイント）。
595    /// `(x, y)` は 1 行目のベースライン左端。
596    pub fn add_text(&mut self, index: usize, text: &str, opts: &TextOptions) -> Result<()> {
597        let font_name = self.ensure_standard_font(index, opts.font)?;
598        let leading = opts.leading.unwrap_or(opts.size * 1.2);
599        let (r, g, b) = opts.color;
600
601        let mut ops = vec![
602            Operation::new("q", vec![]),
603            Operation::new("BT", vec![]),
604            Operation::new("Tf", vec![Object::name(font_name), opts.size.into()]),
605            Operation::new("rg", vec![r.into(), g.into(), b.into()]),
606            Operation::new("TL", vec![leading.into()]),
607            Operation::new("Td", vec![opts.x.into(), opts.y.into()]),
608        ];
609        for (i, line) in text.split('\n').enumerate() {
610            if i > 0 {
611                ops.push(Operation::new("T*", vec![]));
612            }
613            let encoded: Vec<u8> = line
614                .chars()
615                .map(|c| crate::font::char_to_winansi(c).unwrap_or(b'?'))
616                .collect();
617            ops.push(Operation::new("Tj", vec![Object::string_literal(encoded)]));
618        }
619        ops.push(Operation::new("ET", vec![]));
620        ops.push(Operation::new("Q", vec![]));
621        self.append_content(index, &ops)
622    }
623
624    /// ページに直線を描画する。
625    pub fn draw_line(
626        &mut self,
627        index: usize,
628        from: (f64, f64),
629        to: (f64, f64),
630        opts: &DrawOptions,
631    ) -> Result<()> {
632        let (r, g, b) = opts.stroke_color;
633        let ops = vec![
634            Operation::new("q", vec![]),
635            Operation::new("w", vec![opts.line_width.into()]),
636            Operation::new("RG", vec![r.into(), g.into(), b.into()]),
637            Operation::new("m", vec![from.0.into(), from.1.into()]),
638            Operation::new("l", vec![to.0.into(), to.1.into()]),
639            Operation::new("S", vec![]),
640            Operation::new("Q", vec![]),
641        ];
642        self.append_content(index, &ops)
643    }
644
645    /// ページに矩形を描画する。`fill_color` が `Some` なら塗り潰し。
646    pub fn draw_rect(
647        &mut self,
648        index: usize,
649        x: f64,
650        y: f64,
651        width: f64,
652        height: f64,
653        opts: &DrawOptions,
654    ) -> Result<()> {
655        let (r, g, b) = opts.stroke_color;
656        let mut ops = vec![
657            Operation::new("q", vec![]),
658            Operation::new("w", vec![opts.line_width.into()]),
659            Operation::new("RG", vec![r.into(), g.into(), b.into()]),
660        ];
661        let paint = if let Some((fr, fg, fb)) = opts.fill_color {
662            ops.push(Operation::new("rg", vec![fr.into(), fg.into(), fb.into()]));
663            "B" // 塗り + 線
664        } else {
665            "S" // 線のみ
666        };
667        ops.push(Operation::new(
668            "re",
669            vec![x.into(), y.into(), width.into(), height.into()],
670        ));
671        ops.push(Operation::new(paint, vec![]));
672        ops.push(Operation::new("Q", vec![]));
673        self.append_content(index, &ops)
674    }
675}

737        let obj = self.info()?.get(key)?;

741        }

746        let encoded = encode_text_string(value);
747        let info_id = match self.trailer.get("Info").and_then(|o| o.as_reference().ok()) {
748            Some(id) if self.objects.contains_key(&id) => id,
749            _ => {
750                let id = self.add_object(Dictionary::new());
751                self.trailer.set("Info", Object::Reference(id));
752                id
753            }
754        };
755        let info = self.get_object_mut(info_id)?.as_dict_mut()?;
756        info.set(key, Object::String(encoded, crate::object::StringFormat::Literal));
757        Ok(())
758    }

763    }

788// ---------------------------------------------------------------------------
789// テキスト文字列（§7.9.2.2）のエンコード/デコード
790// ---------------------------------------------------------------------------
791
792/// PDF テキスト文字列をデコードする。
793///
794/// UTF-16BE（BOM `FE FF`）、UTF-8（BOM `EF BB BF`, PDF 2.0）、
795/// それ以外は PDFDocEncoding（おおむね Latin-1）として解釈する。
796pub fn decode_text_string(bytes: &[u8]) -> String {
797    if bytes.starts_with(&[0xFE, 0xFF]) {
798        // UTF-16BE
799        let units: Vec<u16> = bytes[2..]
800            .chunks_exact(2)
801            .map(|c| u16::from_be_bytes([c[0], c[1]]))
802            .collect();
803        String::from_utf16_lossy(&units)
804    } else if bytes.starts_with(&[0xEF, 0xBB, 0xBF]) {
805        String::from_utf8_lossy(&bytes[3..]).into_owned()
806    } else {
807        bytes.iter().map(|&b| crate::font::winansi_to_char(b)).collect()
808    }
809}
810
811/// 文字列を PDF テキスト文字列にエンコードする。
812///
813/// ASCII のみなら素のバイト列、それ以外は BOM 付き UTF-16BE。
814pub fn encode_text_string(s: &str) -> Vec<u8> {
815    if s.is_ascii() {
816        s.as_bytes().to_vec()
817    } else {
818        let mut out = vec![0xFE, 0xFF];
819        for unit in s.encode_utf16() {
820            out.extend_from_slice(&unit.to_be_bytes());
821        }
822        out
823    }
824}
825
826fn parse_header_version(data: &[u8]) -> Result<String> {
827    let head = &data[..data.len().min(1024)];
828    let pos = head
829        .windows(5)
830        .position(|w| w == b"%PDF-")
831        .ok_or(PdfError::NotAPdf)?;
832    let rest = &data[pos + 5..];
833    let end = rest
834        .iter()
835        .position(|&b| !(b.is_ascii_digit() || b == b'.'))
836        .unwrap_or(rest.len());
837    let v = String::from_utf8_lossy(&rest[..end]).into_owned();
838    if v.is_empty() {
839        return Err(PdfError::NotAPdf);
840    }
841    Ok(v)
842}
