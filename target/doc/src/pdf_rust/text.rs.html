1//! テキスト抽出。
2//!
3//! コンテントストリームのテキスト表示演算子（`Tj` `TJ` `'` `"`）を解釈し、
4//! フォントのエンコーディング情報を使ってバイト列を Unicode に変換する。
5//!
6//! 対応している変換手段（優先順）:
7//! 1. フォントの `/ToUnicode` CMap（`bfchar` / `bfrange`）
8//! 2. 1 バイトフォントの WinAnsiEncoding（≒ CP1252）フォールバック
9//!
10//! 改行・空白は演算子（`Td` `TD` `T*` `Tm` と `TJ` 内の字送り調整）から
11//! ヒューリスティックに復元する。レイアウトの完全な再現は目的としない。
12

21/// 1 フォント分のデコード情報。
22struct FontDecoder {
23    /// コード長が 2 バイト（Type0/CID フォント）か。
24    two_byte: bool,
25    /// `/ToUnicode` CMap から構築した コード → 文字列 のマップ。
26    to_unicode: Option<HashMap<u32, String>>,
27}
28
29impl FontDecoder {
30    /// フォント辞書から構築する。
31    fn from_font_dict(doc: &Document, font: &Dictionary) -> FontDecoder {
32        let subtype = font.get("Subtype").and_then(|o| o.as_name().ok()).unwrap_or("");
33        let two_byte = subtype == "Type0";
34        let to_unicode = doc
35            .dict_get(font, "ToUnicode")
36            .and_then(|o| o.as_stream().ok())
37            .and_then(|s| doc.get_stream_data(s).ok())
38            .map(|data| parse_tounicode_cmap(&data));
39        FontDecoder { two_byte, to_unicode }
40    }
41
42    /// 表示文字列のバイト列を Unicode 文字列へ変換する。
43    fn decode(&self, bytes: &[u8], out: &mut String) {
44        if self.two_byte {
45            for pair in bytes.chunks(2) {
46                let code = if pair.len() == 2 {
47                    u16::from_be_bytes([pair[0], pair[1]]) as u32
48                } else {
49                    pair[0] as u32
50                };
51                match self.to_unicode.as_ref().and_then(|m| m.get(&code)) {
52                    Some(s) => out.push_str(s),
53                    None => out.push('\u{FFFD}'), // マップ不明の CID
54                }
55            }
56        } else {
57            for &b in bytes {
58                match self.to_unicode.as_ref().and_then(|m| m.get(&(b as u32))) {
59                    Some(s) => out.push_str(s),
60                    None => out.push(crate::font::winansi_to_char(b)),
61                }
62            }
63        }
64    }
65}

67/// ページのテキストを抽出する。
68pub fn extract_page_text(doc: &Document, page_id: ObjectId) -> Result<String> {
69    let content = doc.page_content_bytes(page_id)?;
70    let resources = doc.page_resources(page_id);
71    let mut out = String::new();
72    extract_from_content(doc, &content, &resources, &mut out, 0)?;
73    // 末尾の余分な改行を整理
74    while out.ends_with('\n') {
75        out.pop();
76    }
77    Ok(out)
78}

80/// コンテントストリーム 1 本分を処理する。Form XObject は再帰する。
81fn extract_from_content(
82    doc: &Document,
83    content: &[u8],
84    resources: &Dictionary,
85    out: &mut String,
86    depth: usize,
87) -> Result<()> {
88    if depth > 8 {
89        return Ok(()); // フォーム再帰の暴走防止
90    }
91    let ops = parse_content(content)?;
92
93    // リソースからフォント辞書を引く（遅延構築でキャッシュ）
94    let fonts_dict = doc
95        .dict_get(resources, "Font")
96        .and_then(|o| o.as_dict().ok())
97        .cloned()
98        .unwrap_or_default();
99    let mut decoders: HashMap<String, FontDecoder> = HashMap::new();
100    let mut current_font: Option<String> = None;
101
102    // 行検出用の状態
103    let mut last_ty: Option<f64> = None; // Tm の縦位置
104    let mut pending_newline = false;
105
106    let push_newline = |out: &mut String| {
107        if !out.is_empty() && !out.ends_with('\n') {
108            out.push('\n');
109        }
110    };
111
112    for op in &ops {
113        match op.operator.as_str() {
114            "BT" => {
115                last_ty = None;
116            }
117            "Tf" => {
118                if let Some(Object::Name(n)) = op.operands.first() {
119                    if !decoders.contains_key(n) {
120                        let decoder = doc
121                            .dict_get(&fonts_dict, n)
122                            .and_then(|o| o.as_dict().ok())
123                            .map(|fd| FontDecoder::from_font_dict(doc, fd))
124                            .unwrap_or(FontDecoder { two_byte: false, to_unicode: None });
125                        decoders.insert(n.clone(), decoder);
126                    }
127                    current_font = Some(n.clone());
128                }
129            }
130            // 行送り系: 縦移動があれば改行
131            "Td" | "TD" => {
132                if let Some(ty) = op.operands.get(1).and_then(|o| o.as_number().ok()) {
133                    if ty != 0.0 {
134                        pending_newline = true;
135                    } else if let Some(tx) = op.operands.first().and_then(|o| o.as_number().ok()) {
136                        if tx > 0.0 && !out.is_empty() && !out.ends_with(char::is_whitespace) {
137                            out.push(' ');
138                        }
139                    }
140                }
141            }
142            "T*" => pending_newline = true,
143            "Tm" => {
144                if let Some(f) = op.operands.get(5).and_then(|o| o.as_number().ok()) {
145                    if let Some(prev) = last_ty {
146                        if (f - prev).abs() > 0.5 {
147                            pending_newline = true;
148                        }
149                    }
150                    last_ty = Some(f);
151                }
152            }
153            "Tj" | "'" | "\"" => {
154                if op.operator != "Tj" {
155                    pending_newline = true; // ' と " は次行へ移ってから表示
156                }
157                if pending_newline {
158                    push_newline(out);
159                    pending_newline = false;
160                }
161                // " のオペランドは (aw ac string)
162                let s = op.operands.iter().rev().find_map(|o| o.as_string().ok());
163                if let Some(bytes) = s {
164                    decode_with(&decoders, &current_font, bytes, out);
165                }
166            }
167            "TJ" => {
168                if pending_newline {
169                    push_newline(out);
170                    pending_newline = false;
171                }
172                if let Some(Object::Array(items)) = op.operands.first() {
173                    for item in items {
174                        match item {
175                            Object::String(bytes, _) => {
176                                decode_with(&decoders, &current_font, bytes, out)
177                            }
178                            // 大きな負の字送り調整は単語間の空白とみなす
179                            Object::Integer(_) | Object::Real(_) => {
180                                let adj = item.as_number().unwrap_or(0.0);
181                                if adj < -180.0
182                                    && !out.is_empty()
183                                    && !out.ends_with(char::is_whitespace)
184                                {
185                                    out.push(' ');
186                                }
187                            }
188                            _ => {}
189                        }
190                    }
191                }
192            }
193            // Form XObject の再帰
194            "Do" => {
195                if let Some(Object::Name(n)) = op.operands.first() {
196                    let xobj = doc
197                        .dict_get(resources, "XObject")
198                        .and_then(|o| o.as_dict().ok())
199                        .and_then(|x| doc.dict_get(x, n))
200                        .and_then(|o| o.as_stream().ok().cloned());
201                    if let Some(stream) = xobj {
202                        let is_form = stream
203                            .dict
204                            .get("Subtype")
205                            .and_then(|o| o.as_name().ok())
206                            == Some("Form");
207                        if is_form {
208                            if let Ok(data) = doc.get_stream_data(&stream) {
209                                let sub_res = match doc.dict_get(&stream.dict, "Resources") {
210                                    Some(Object::Dictionary(d)) => d.clone(),
211                                    _ => resources.clone(),
212                                };
213                                extract_from_content(doc, &data, &sub_res, out, depth + 1)?;
214                            }
215                        }
216                    }
217                }
218            }
219            "ET" => {
220                pending_newline = false;
221                push_newline(out);
222            }
223            _ => {}
224        }
225    }
226    Ok(())
227}

229fn decode_with(
230    decoders: &HashMap<String, FontDecoder>,
231    current: &Option<String>,
232    bytes: &[u8],
233    out: &mut String,
234) {
235    static FALLBACK: std::sync::OnceLock<FontDecoder> = std::sync::OnceLock::new();
236    let fallback = FALLBACK.get_or_init(|| FontDecoder { two_byte: false, to_unicode: None });
237    let decoder = current.as_ref().and_then(|n| decoders.get(n)).unwrap_or(fallback);
238    decoder.decode(bytes, out);
239}

241// ---------------------------------------------------------------------------
242// ToUnicode CMap の解析（§9.10.3）
243// ---------------------------------------------------------------------------
244
245/// ToUnicode CMap ストリームから コード → Unicode 文字列 マップを作る。
246///
247/// `beginbfchar`/`endbfchar` と `beginbfrange`/`endbfrange` のみ解釈する
248/// （ToUnicode 用途ではこれで十分）。
249fn parse_tounicode_cmap(data: &[u8]) -> HashMap<u32, String> {
250    let mut map = HashMap::new();
251    let mut lexer = Lexer::new(data);
252    // 直近のオペランドを溜めるスタック
253    let mut stack: Vec<Token> = Vec::new();
254    loop {
255        let token = match lexer.next_token() {
256            Ok(Token::Eof) => break,
257            Ok(t) => t,
258            Err(_) => continue,
259        };
260        match &token {
261            Token::Keyword(k) if k == "beginbfchar" => {
262                stack.clear();
263                // <src> <dst> の対が endbfchar まで続く
264                while let Ok(Token::HexString(src)) = lexer.next_token() {
265                    match lexer.next_token() {
266                        Ok(Token::HexString(dst)) => {
267                            if let Some(code) = bytes_to_code(&src) {
268                                map.insert(code, utf16be_to_string(&dst));
269                            }
270                        }
271                        _ => break,
272                    }
273                }
274            }
275            Token::Keyword(k) if k == "beginbfrange" => {
276                stack.clear();
277                // <lo> <hi> <dst...> の組が endbfrange まで続く
278                while let Ok(Token::HexString(lo)) = lexer.next_token() {
279                    let hi = match lexer.next_token() {
280                        Ok(Token::HexString(s)) => s,
281                        _ => break,
282                    };
283                    let (lo_c, hi_c) = match (bytes_to_code(&lo), bytes_to_code(&hi)) {
284                        (Some(a), Some(b)) if b >= a && b - a < 65536 => (a, b),
285                        _ => break,
286                    };
287                    match lexer.next_token() {
288                        // 連続マッピング: <lo> <hi> <dstStart>
289                        Ok(Token::HexString(dst)) => {
290                            for i in 0..=(hi_c - lo_c) {
291                                let mut d = dst.clone();
292                                // 最後の 16bit 単位に加算
293                                if d.len() >= 2 {
294                                    let l = d.len();
295                                    let unit = u16::from_be_bytes([d[l - 2], d[l - 1]])
296                                        .wrapping_add(i as u16);
297                                    d[l - 2..].copy_from_slice(&unit.to_be_bytes());
298                                }
299                                map.insert(lo_c + i, utf16be_to_string(&d));
300                            }
301                        }
302                        // 個別マッピング: <lo> <hi> [<d0> <d1> ...]
303                        Ok(Token::ArrayStart) => {
304                            let mut i = 0u32;
305                            loop {
306                                match lexer.next_token() {
307                                    Ok(Token::HexString(d)) => {
308                                        map.insert(lo_c + i, utf16be_to_string(&d));
309                                        i += 1;
310                                    }
311                                    Ok(Token::ArrayEnd) => break,
312                                    _ => break,
313                                }
314                            }
315                        }
316                        _ => break,
317                    }
318                }
319            }
320            t => {
321                stack.push(t.clone());
322                if stack.len() > 16 {
323                    stack.remove(0);
324                }
325            }
326        }
327    }
328    map
329}

331/// CMap のコード（1〜4 バイトのビッグエンディアン）を u32 へ。
332fn bytes_to_code(bytes: &[u8]) -> Option<u32> {
333    if bytes.is_empty() || bytes.len() > 4 {
334        return None;
335    }
336    Some(bytes.iter().fold(0u32, |acc, &b| (acc << 8) | b as u32))
337}

339/// UTF-16BE のバイト列を文字列へ（サロゲートペア対応）。
340fn utf16be_to_string(bytes: &[u8]) -> String {
341    let units: Vec<u16> =
342        bytes.chunks_exact(2).map(|c| u16::from_be_bytes([c[0], c[1]])).collect();
343    String::from_utf16_lossy(&units)
344}

346#[cfg(test)]
347mod tests {

350    #[test]
351    fn parses_bfchar_and_bfrange() {
352        let cmap = br"
353/CIDInit /ProcSet findresource begin
35412 dict begin begincmap
3551 begincodespacerange <0000> <FFFF> endcodespacerange
3562 beginbfchar
357<0041> <0061>
358<0042> <3042>
359endbfchar
3601 beginbfrange
361<0050> <0052> <0070>
362endbfrange
363endcmap end end";
364        let map = parse_tounicode_cmap(cmap);
365        assert_eq!(map[&0x41], "a");
366        assert_eq!(map[&0x42], "あ");
367        assert_eq!(map[&0x50], "p");
368        assert_eq!(map[&0x51], "q");
369        assert_eq!(map[&0x52], "r");
370    }
371
372    #[test]
373    fn bfrange_with_array() {
374        let cmap = b"1 beginbfrange <01> <03> [<0058> <0059> <005A>] endbfrange";
375        let map = parse_tounicode_cmap(cmap);
376        assert_eq!(map[&1], "X");
377        assert_eq!(map[&2], "Y");
378        assert_eq!(map[&3], "Z");
379    }
380
381    #[test]
382    fn surrogate_pair_target() {
383        // U+1F600 (😀) は UTF-16BE で D83D DE00
384        let cmap = b"1 beginbfchar <01> <D83DDE00> endbfchar";
385        let map = parse_tounicode_cmap(cmap);
386        assert_eq!(map[&1], "\u{1F600}");
387    }
